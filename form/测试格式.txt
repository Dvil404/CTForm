server {

    listen 80;
    listen 443 ssl;

    ssl_certificate /root/server.crt; ## 第6步⾥⽣成的 server.crt ⽂件所在位置
    ssl_certificate_key /root/server.key; ## 第6步⾥⽣成的 server.key⽂件所在位置

    server_name 44.44.44.44;
    proxy_cookie_path / "/; secure";
    
location / {
    
    proxy_pass http://127.0.0.1:7001;

}
}
